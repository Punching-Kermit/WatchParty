const video = document.getElementById('main-video'); const status = document.getElementById('status'); const friendInput = document.getElementById('friend-id'); window.onload = () => { const savedID = localStorage.getItem('watchPartyFriendID'); if(savedID) { friendInput.value = savedID; status.innerText = "Saved ID loaded."; } }; const peer = new Peer({ host: '0.peerjs.com', port: 443, secure: true }); let conn; peer.on('open', id => { document.getElementById('my-id').innerText = "YOUR ID: " + id; }); peer.on('connection', c => { conn = c; status.innerText = "Friend Joined!"; setupConn(); }); function connectToFriend() { const fID = friendInput.value.trim(); localStorage.setItem('watchPartyFriendID', fID); conn = peer.connect(fID); conn.on('open', () => { status.innerText = "Connected!"; setupConn(); }); } function setupConn() { conn.on('data', async (data) => { if (data.type === 'sync') { video.src = data.url; if (data.subContent) { const blob = new Blob([data.subContent], { type: 'text/vtt' }); applySubtitles(URL.createObjectURL(blob)); } video.load(); } else if (data.type === 'play') { video.currentTime = data.time; video.play(); } else if (data.type === 'pause') { video.pause(); } }); } function applySubtitles(url) { const old = video.querySelectorAll('track'); old.forEach(t => t.remove()); const track = document.createElement('track'); Object.assign(track, { kind: 'subtitles', label: 'English', srclang: 'en', src: url, default: true }); video.appendChild(track); } async function syncAll() { const mUrl = document.getElementById('movie-url').value; const sUrl = document.getElementById('sub-url').value; let subContent = ""; // This part "grabs" the text from the link so we don't rely on the link later if (sUrl) { try { const response = await fetch(sUrl); subContent = await response.text(); } catch (e) { console.log("Subtitle fetch failed, using link only."); } } video.src = mUrl; if (subContent) { const blob = new Blob([subContent], { type: 'text/vtt' }); applySubtitles(URL.createObjectURL(blob)); } video.load(); if (conn && conn.open) { conn.send({ type: 'sync', url: mUrl, subContent: subContent }); } } video.onplay = () => { if (conn && conn.open) conn.send({ type: 'play', time: video.currentTime }); }; video.onpause = () => { if (conn && conn.open) conn.send({ type: 'pause' }); };
